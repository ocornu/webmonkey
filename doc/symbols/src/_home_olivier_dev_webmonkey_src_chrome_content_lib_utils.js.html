 13 function GM_hitch(obj, meth) {
 14   if (!obj[meth]) {
 15     throw "method '" + meth + "' does not exist on object '" + obj + "'";
 16   }
 17 
 18   var staticArgs = Array.prototype.splice.call(arguments, 2, arguments.length);
 19 
 20   return function() {
 21     // make a copy of staticArgs (don't modify it because it gets reused for
 22     // every invocation).
 23     var args = staticArgs.concat();
 24 
 25     // add all the new arguments
 26     for (var i = 0; i < arguments.length; i++) {
 27       args.push(arguments[i]);
 28     }
 29 
 30     // invoke the original function with the correct this obj and the combined
 31     // list of static and dynamic arguments.
 32     return obj[meth].apply(obj, args);
 33   };
 34 }

 61 /**
 62  * Takes the place of the traditional prompt() function which became broken
 63  * in FF 1.0.1. :(
 64  */
 65 function gmPrompt(msg, defVal, title) {
 66   var promptService = Components.classes["@mozilla.org/embedcomp/prompt-service;1"]
 67                                 .getService(Components.interfaces.nsIPromptService);
 68   var result = {value:defVal};
 69 
 70   if (promptService.prompt(null, title, msg, result, null, {value:0})) {
 71     return result.value;
 72   }
 73   else {
 74     return null;
 75   }
 76 }

 78 function dbg(o) {
 79   var s = "";
 80   var i = 0;
 81 
 82   for (var p in o) {
 83     s += p + ":" + o[p] + "\n";
 84 
 85     if (++i % 15 == 0) {
 86       alert(s);
 87       s = "";
 88     }
 89   }
 90 
 91   alert(s);
 92 }

102 function GM_isGreasemonkeyable(url) {
103   var scheme = Components.classes["@mozilla.org/network/io-service;1"]
104                .getService(Components.interfaces.nsIIOService)
105                .extractScheme(url);
106 
107   return (scheme == "http" || scheme == "https" || scheme == "file" ||
108           scheme == "ftp" || url.match(/^about:cache/)) &&
109           !/hiddenWindow\.html$/.test(url);
110 }

139 /**
140  * Loggifies an object. Every method of the object will have it's entrance,
141  * any parameters, any errors, and it's exit logged automatically.
142  */
143 function loggify(obj, name) {
144   for (var p in obj) {
145     if (typeof obj[p] == "function") {
146       obj[p] = gen_loggify_wrapper(obj[p], name, p);
147     }
148   }
149 }
150 
151 function gen_loggify_wrapper(meth, objName, methName) {
152   return function() {
153      var retVal;
154     //var args = new Array(arguments.length);
155     var argString = "";
156     for (var i = 0; i < arguments.length; i++) {
157       //args[i] = arguments[i];
158       argString += arguments[i] + (((i+1)<arguments.length)? ", " : "");
159     }
160 
161     log("> %s.%s(%s)", objName, methName, argString); //args.join(", "));
162 
163     try {
164       return retVal = meth.apply(this, arguments);
165     } finally {
166       log("< %s.%s: %s",
167           objName,
168           methName,
169           (typeof retVal == "undefined" ? "void" : retVal));
170     }
171   }
172 }
