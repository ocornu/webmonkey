  3 var bundle = null;
  4 window.addEventListener("load", function() {
  5   // init the global string bundle
  6   bundle = document.getElementById("gm-browser-bundle");
  7 
  8   // load default namespace from pref
  9   document.getElementById("namespace").value =
 10       GM_prefRoot.get("newscript_namespace", "");
 11 
 12   // default the includes with the current page's url
 13   document.getElementById("includes").value =
 14       window.opener.document.getElementById("content").selectedBrowser
 15       .contentWindow.location.href;
 16 }, false);

 20 function doInstall() {
 21   var source = createScriptSource();
 22   if (!source) return false;
 23   var script = Script.fromSource(source);
 24   // persist namespace value
 25   GM_prefRoot.set("newscript_namespace", script.namespace);
 26 
 27   var config = GM_getConfig();
 28   // make sure entered details will not ruin an existing file
 29   if (config.installIsUpdate(script)) {
 30     var overwrite = confirm(bundle.getString("newscript.exists"));
 31     if (!overwrite) return false;
 32   }
 33   config.install(script);
 34   script.edit(window);
 35   return true;
 36 }

 38 // assemble the XUL fields into a script template
 39 function createScriptSource() {
 40   var source = ["// ==UserScript=="];
 41 
 42   var name = document.getElementById("name").value;
 43   if (name == "") {
 44     alert(bundle.getString("newscript.noname"));
 45     return false;
 46   }
 47   source.push("// @name           " + name);
 48 
 49   var namespace = document.getElementById("namespace").value;
 50   if (namespace == "") {
 51     alert(bundle.getString("newscript.nonamespace"));
 52     return false;
 53   }
 54   source.push("// @namespace      " + namespace);
 55 
 56   var descr = document.getElementById("descr").value;
 57   if (descr != "")
 58     source.push("// @description    " + descr);
 59 
 60   var includes = document.getElementById("includes").value;
 61   if (includes != "") {
 62     includes = includes.match(/.+/g);
 63     includes = "// @include        " + includes.join("\n// @include        ");
 64     source.push(includes);
 65   }
 66 
 67   var excludes = document.getElementById("excludes").value;
 68   if (excludes != "") {
 69     excludes = excludes.match(/.+/g);
 70     excludes = "// @exclude        " + excludes.join("\n// @exclude        ");
 71     source.push(excludes);
 72   }
 73 
 74   source.push("// ==/UserScript==");
 75   return source.join("\n");
 76 }
