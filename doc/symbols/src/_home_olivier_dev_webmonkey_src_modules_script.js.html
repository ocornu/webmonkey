284   var script = new Script();

434   parse: function(script, source, uri) {
435     var meta = false;
436     for each (var line in source.match(/.+/g)) {
437       if (!meta) {
438         if (line.indexOf("// ==UserScript==") == 0) meta = true;
439         continue;
440       }
441       if (line.indexOf("// ==/UserScript==") == 0) break;
442       var match = line.match(/\/\/ \@(\S+)(?:\s+([^\n]+))?/);
443       if (match === null) continue;
444 
445       var header = match[1];
446       var value  = match[2];
447       if (value)          // @header <value>
448         switch (header) {
449         case "name":
450         case "namespace":
451         case "description":
452           this[header] = value;
453           break;
454         case "include":
455         case "exclude":
456           this[header].push(value);
457           break;
458         case "require":
459           var require = new Script.Require(script);
460           require.parse(value, uri);
461           this.require.push(require);
462           break;
463         case "resource":
464           var resource = new Script.Resource(script);
465           resource.parse(value, uri);
466           this.resource.push(resource);
467           break;
468         }
469     }
470     
471     // if no meta info, default to reasonable values
472     if (this.name == null) this.name = parseScriptName(uri);
473     if (this.namespace == null) this.namespace = uri.host;
474     if (!this.description) this.description = "";
475     if (!this.include.length) this.include.push("*");
476 
477     function parseScriptName(sourceUri) {
478       var name = sourceUri.spec;
479       name = name.substring(0, name.indexOf(".user.js"));
480       return name.substring(name.lastIndexOf("/") + 1);
481     }
482   },

831       // chrome:// URLs and the greasemonkey.js component's file:// URL.
832       if (stack.language == 2

961 Script.Api.XmlHttpRequest = function(/**nsIDOMWindow*/ unsafeWin) {
962   this.unsafeWin = unsafeWin;
963   this.chromeWin = SHELL.hiddenDOMWindow;
964 }
965 Script.Api.XmlHttpRequest.prototype = {
966   //this function gets called by user scripts in content security scope to
967   //start a cross-domain xmlhttp request.
968   //
969   //details should look like:
970   //{method,url,onload,onerror,onreadystatechange,headers,data}
971   //headers should be in the form {name:value,name:value,etc}
972   //can't support mimetype because i think it's only used for forcing
973   //text/xml and we can't support that
974   contentStartRequest: function(details) {
975     // important to store this locally so that content cannot trick us up with
976     // a fancy getter that checks the number of times it has been accessed,
977     // returning a dangerous URL the time that we actually use it.
978     var url = details.url;
979     if (typeof url != "string")
980       throw new Error("Invalid url: url must be of type string");
981     var scheme = IO.extractScheme(url);
982     switch (scheme) {
983       case "http":
984       case "https":
985       case "ftp":
986         this.chromeWin.setTimeout(
987           bind(this, "_chromeStartRequest", url, details), 0);
988         break;
989       default:
990         throw new Error("Invalid url: " + url);
991     }
992   },
993 
994   // this function is intended to be called in chrome's security context, so
995   // that it can access other domains without security warning
996   _chromeStartRequest: function(safeUrl, details) {
997     var req = new this.chromeWin.XMLHttpRequest();
998     this._setupRequestEvent(this.unsafeWin, req, "onload", details);
999     this._setupRequestEvent(this.unsafeWin, req, "onerror", details);
1000     this._setupRequestEvent(this.unsafeWin, req, "onreadystatechange", details);
1001     req.open(details.method, safeUrl);
1002     if (details.overrideMimeType)
1003       req.overrideMimeType(details.overrideMimeType);
1004   
1005     if (details.headers)
1006       for (var prop in details.headers)
1007         req.setRequestHeader(prop, details.headers[prop]);
1008   
1009     if (details.nocache)
1010       try {
1011         req.channel.loadFlags |= Ci.nsIRequest.LOAD_BYPASS_CACHE;
1012       } catch (e) {
1013         throw new Error("Could not set 'bypass cache' option");
1014       }
1015   
1016     var body = details.data ? details.data : null;
1017     if (details.binary)
1018       req.sendAsBinary(body);
1019     else
1020       req.send(body);
1021   },
1022 
1023   // arranges for the specified 'event' on xmlhttprequest 'req' to call the
1024   // method by the same name which is a property of 'details' in the content
1025   // window's security context.
1026   _setupRequestEvent: function(unsafeWin, req, event, details) {
1027     if (!details[event])
1028       return;
1029     req[event] = function() {
1030       var responseState = {
1031         // can't support responseXML because security won't
1032         // let the browser call properties on it
1033         responseText:    req.responseText,
1034         readyState:      req.readyState,
1035         responseHeaders:(req.readyState == 4 && event != "onerror" ?
1036                          req.getAllResponseHeaders() :
1037                          ""),
1038         status:         (req.readyState == 4 ? req.status : 0),
1039         statusText:     (req.readyState == 4 && event != "onerror" ?
1040                         req.statusText : ""),
1041         finalUrl:       (req.readyState == 4 ? req.channel.URI.spec : "")
1042       }
1043       // Pop back onto browser thread and call event handler.
1044       // Have to use nested function here instead of GM_hitch because
1045       // otherwise details[event].apply can point to window.setTimeout, which
1046       // can be abused to get increased priveleges.
1047       new XPCNativeWrapper(unsafeWin, "setTimeout()")
1048         .setTimeout(function() { details[event](responseState); }, 0);
1049     }
1050   }

1070 /**
1071  * Transform a script name or a URI into a file name.<br>
1072  * A set of spaces/tabs becomes an underscore, non-Latin chars are removed.
1073  * Latin letters are lower-cased. Numbers, dots and minus sign are allowed. 
1074  * @param aOrigin               What to transform.
1075  * @param [aDefault="noname"]   Default value, in case of empty string.
1076  * @return {string}             A file name.
1077  * @private
1078  */
1079 function toFilename(/**string|nsIURI*/ aOrigin, /**string*/ aDefault) {
1080   var name = aOrigin;
1081   if (typeof aOrigin != "string") {
1082     name = aOrigin.path;
1083     name = name.replace(/^.*\/([^/]*)$/, "$1").replace(/^([^?]*)\?.*$/, "$1");
1084   }
1085   name = name.toLowerCase().replace(/\s+/g, "_").replace(/[^-_A-Z0-9.]+/gi, "");
1086   // If no Latin characters found - use default
1087   if (name.length == 0) name = aDefault ? aDefault : "noname";
1088   return name;
1089 };
