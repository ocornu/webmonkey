1 function GM_ScriptStorage(script) {
  2   this.prefMan = GM_prefRoot.subManager(
  3                  "scriptvals." + script.namespace + "/" + script.name);
  4 }
  5 
  6 GM_ScriptStorage.prototype.setValue = function(name, val) {
  7   if (!GM_apiLeakCheck("GM_setValue")) {
  8     return;
  9   }
 10 
 11   this.prefMan.set(name, val);
 12 };
 13 
 14 GM_ScriptStorage.prototype.getValue = function(name, defVal) {
 15   if (!GM_apiLeakCheck("GM_getValue")) {
 16     return;
 17   }
 18 
 19   return this.prefMan.get(name, defVal);
 20 };

 22 function GM_Resources(script){
 23   this.script = script;
 24 }
 25 
 26 GM_Resources.prototype.getResourceURL = function(name) {
 27   if (!GM_apiLeakCheck("GM_getResourceURL")) {
 28     return;
 29   }
 30 
 31   return this.getDep_(name).dataContent;
 32 };
 33 
 34 GM_Resources.prototype.getResourceText = function(name) {
 35   if (!GM_apiLeakCheck("GM_getResourceText")) {
 36     return;
 37   }
 38 
 39   return this.getDep_(name).textContent;
 40 };
 41 
 42 GM_Resources.prototype.getDep_ = function(name) {
 43   var resources = this.script.resources;
 44   for (var i = 0, resource; resource = resources[i]; i++)
 45     if (resource.name == name)
 46       return resource;
 47   throw new Error("No resource with name: " + name); // NOTE: Non localised string
 48 };

 50 function GM_ScriptLogger(script) {
 51   var namespace = script.namespace;
 52 
 53   if (namespace.substring(namespace.length - 1) != "/") {
 54     namespace += "/";
 55   }
 56 
 57   this.prefix = [namespace, script.name, ": "].join("");
 58 }
 59 
 60 GM_ScriptLogger.prototype.log = function(message) {
 61   GM_log(this.prefix + message, true);
 62 };
 63 
 64 GM_ScriptStorage.prototype.deleteValue = function(name) {
 65   return this.prefMan.remove(name);
 66 }
 67 
 68 GM_ScriptStorage.prototype.listValues = function() {
 69   return this.prefMan.list();
 70 }

 72 // Based on Mark Pilgrim's GM_addGlobalStyle from
 73 // http://diveintogreasemonkey.org/patterns/add-css.html. Used by permission

 75 function GM_addStyle(doc, css) {
 76   var head, style;
 77   head = doc.getElementsByTagName("head")[0];
 78   if (!head) { return; }
 79   style = doc.createElement("style");
 80   style.type = "text/css";
 81   style.innerHTML = css;
 82   head.appendChild(style);
 83 }

 85 function GM_console(script) {
 86   // based on http://www.getfirebug.com/firebug/firebugx.js
 87   var names = [
 88     "debug", "warn", "error", "info", "assert", "dir", "dirxml",
 89     "group", "groupEnd", "time", "timeEnd", "count", "trace", "profile",
 90     "profileEnd"
 91   ];
 92 
 93   for (var i=0, name; name=names[i]; i++) {
 94     this[name] = function() {};
 95   }
 96 
 97   // Important to use this private variable so that user scripts can't make
 98   // this call something else by redefining <this> or <logger>.
 99   var logger = new GM_ScriptLogger(script);
100   this.log = function() {
101     logger.log(
102       Array.prototype.slice.apply(arguments).join("\n")
103     );
104   };
105 }
