 72    * <code>nsIObserver</code> event handler.

 75   observe: function(aSubject, aTopic, aData) {
 76     if (aTopic != "app-startup") return;
 77     Components.utils.import("resource://webmonkey/config.js");

 81     loader.loadSubScript("chrome://webmonkey/content/lib/utils.js");

 86   /**
 87    * <code>nsIContentPolicy</code> resource loader hook.
 88    * Allow all resources to be loaded except userscripts, which are redirected
 89    * to the script installer.
 90    * @type  int
 91    */
 92   shouldLoad: function(contentType, contentLocation, requestOrigin, context,
 93                        mimeTypeGuess, extra) {
 94     var ret = Ci.nsIContentPolicy.ACCEPT;
 95 
 96     // block content detection of webmonkey by denying GM
 97     // chrome content, unless loaded from chrome
 98     if (requestOrigin && requestOrigin.scheme != "chrome"
 99         && contentLocation.scheme == "chrome"
100         && contentLocation.host == "webmonkey")
101       return Ci.nsIContentPolicy.REJECT_SERVER;
102 
103     // don't intercept anything when GM is not enabled
104     if (!GM_getEnabled()) return ret;
105 
106     // don't interrupt the view-source: scheme
107     // (triggered if the link in the error console is clicked)
108     if (contentLocation.scheme == "view-source") return ret;
109 
110     if (contentType == Ci.nsIContentPolicy.TYPE_DOCUMENT
111         && contentLocation.spec.match(/\.user\.js$/)) {
112       var winWat = Cc["@mozilla.org/embedcomp/window-watcher;1"]
113                    .getService(Ci.nsIWindowWatcher);
114       if (winWat.activeWindow && winWat.activeWindow.GM_BrowserUI) {
115         winWat.activeWindow.GM_BrowserUI.downloadScript(contentLocation, true);
116         ret = Ci.nsIContentPolicy.REJECT_REQUEST;
117       }
118     }
119 
120     return ret;
121   },

144   /**
145    * <code>DOMContentLoaded</code> event handler.
146    * Inject relevant scripts once DOM is fully loaded.
147    * @param {Window} unsafeWin
148    *        Target DOM window.
149    * @param {ChromeWindow} chromeWin
150    *        Target chrome window.
151    * @param {GM_BrowserUI} gmBrowser
152    *        <code>GM_BrowserUI</code> responsible for this chrome window.
153    */
154   domContentLoaded: function(unsafeWin, chromeWin, gmBrowser) {
155     var safeWin   = new XPCNativeWrapper(unsafeWin);
156     var href      = safeWin.location.href;
157     var scripts   = this.config.getMatchingScripts(
158       function(script) { return script.enabled && script.matchesURL(href); }
159     );
160     // assert there are scripts to inject
161     if (!scripts.length) return;
162 
163     var firebug = getFirebugConsole(safeWin, unsafeWin, chromeWin);
164     for each (var script in scripts)
165       inject(script, safeWin, gmBrowser, firebug);
166 
167     /* FireBug 1.2+ console support */
168     function getFirebugConsole(safeWin, unsafeWin, chromeWin) {
169       try {
170         chromeWin = chromeWin.top;
171         // assert FB is installed
172         if (!chromeWin.Firebug)
173           return null;
174         var fbVersion = parseFloat(chromeWin.Firebug.version);
175         var fbConsole = chromeWin.Firebug.Console;
176         var fbContext = chromeWin.TabWatcher &&
177                         chromeWin.TabWatcher.getContextByWindow(unsafeWin);
178         // assert FB is enabled
179         if (!fbConsole.isEnabled(fbContext))
180           return null;
181 
182         if (fbVersion == 1.2) {
183           // search console handler
184           if (fbContext.consoleHandler)
185             for (var i = 0; i < fbContext.consoleHandler.length; i++)
186               if (fbContext.consoleHandler[i].window == safeWin)
187                 return fbContext.consoleHandler[i].handler;
188           var dummyElm = safeWin.document.createElement("div");
189           dummyElm.setAttribute("id", "_firebugConsole");
190           safeWin.document.documentElement.appendChild(dummyElm);
191           chromeWin.Firebug.Console.injector.addConsoleListener(fbContext, safeWin);
192           dummyElm.parentNode.removeChild(dummyElm);
193           return fbContext.consoleHandler.pop().handler;
194         }
195 
196         if (fbVersion == 1.3 || fbVersion == 1.4) {
197           fbConsole.injector.attachIfNeeded(fbContext, unsafeWin);
198           // find active context
199           for (var i=0; i<fbContext.activeConsoleHandlers.length; i++)
200             if (fbContext.activeConsoleHandlers[i].window == unsafeWin)
201               return fbContext.activeConsoleHandlers[i];
202           return null;
203         }
204       } catch (e) {
205         dump('Webmonkey getFirebugConsole() error:\n'+uneval(e)+'\n');
206       }
207       return null;
208     }
209   }

223 /**
224  * Inject a script into a DOM window.
225  * @param {Script} script
226  *        <code>Script</code> to be injected.
227  * @param {XPCNativeWrapper} safeWin
228  *        Target <code>Window</code> to inject script into.
229  * @param {GM_BrowserUI} gmBrowser
230  *        <code>GM_BrowserUI</code> responsible for this window.
231  * @param fbConsole
232  *        Firebug console.
233  * @return  <code>true</code> if script succeeds, <code>false</code> otherwise.
234  * @type    boolean
235  */
236 function inject(script, safeWin, gmBrowser, fbConsole) {
237   var sandbox   = new Components.utils.Sandbox(safeWin);
238   var logger    = new GM_ScriptLogger(script);
239   var console   = fbConsole ? fbConsole : new GM_console(script);
240   var storage   = new GM_ScriptStorage(script);
241   var unsafeWin = safeWin.wrappedJSObject;
242   var xhr       = new GM_xmlhttpRequester(unsafeWin, appSvc.hiddenDOMWindow);
243   var resources = new GM_Resources(script);
244 
245   // populate sandbox
246   sandbox.window       = safeWin;
247   sandbox.document     = safeWin.document;
248   sandbox.unsafeWindow = unsafeWin;
249   sandbox.console      = console;
250   sandbox.XPathResult  = Ci.nsIDOMXPathResult;
251   // add our own APIs
252   var GM = sandbox.GM = {};
253   GM.addStyle            = function(css) { GM_addStyle(safeDoc, css) };
254   GM.log                 = GM_hitch(logger, "log");
255   GM.setValue            = GM_hitch(storage, "setValue");
256   GM.getValue            = GM_hitch(storage, "getValue");
257   GM.deleteValue         = GM_hitch(storage, "deleteValue");
258   GM.listValues          = GM_hitch(storage, "listValues");
259   GM.getResourceURL      = GM_hitch(resources, "getResourceURL");
260   GM.getResourceText     = GM_hitch(resources, "getResourceText");
261   GM.xmlhttpRequest      = GM_hitch(xhr, "contentStartRequest");
262   GM.openInTab           = GM_hitch(gmBrowser, "openInTab");
263   GM.registerMenuCommand = GM_hitch(gmBrowser, "registerMenuCommand",
264                                     unsafeWin);
265   sandbox.__proto__ = safeWin;
266 
267   // compile @requires
268   var requires = [];
269   var offsets = [];
270   var offset = 0;
271   for each(var req in script.requires) {
272     var contents = req.textContent;
273     var lineCount = contents.split("\n").length;
274     requires.push(contents);
275     offset += lineCount;
276     offsets.push(offset);
277   }
278   script.offsets = offsets;
279 
280   // script source (error line-number calculations depend on these \n)
281   var source = "\n" + requires.join("\n") + "\n" +
282                script.textContent + "\n";
283   var api    = "for (var i in GM) eval('var GM_'+i+' = GM[i]');";
284   if (script.unwrap)
285     source = api+source;
286   else {
287     // move API inside script wrapper
288     api = "const GM = this.GM; delete this.GM; "+ api +"\
289            var window = this.window; delete this.window;\
290            var unsafeWindow = this.unsafeWindow; delete this.unsafeWindow;\
291            var document = this.document; delete this.document;\
292            var XPathResult = this.XPathResult; delete this.XPathResult;\
293            var console = this.console; delete this.console;";
294     // wrap script into an anonymous function
295     source = "(function(){"+ api+source +"})()";
296   }
297 
298   // eval in sandbox
299   try {
300     // workaround for https://bugzilla.mozilla.org/show_bug.cgi?id=307984
301     var lineRef = new Error().lineNumber + 1;
302     Components.utils.evalInSandbox(source, sandbox);
303     return true;
304   } catch (e) {
305     // try to find the line of the actual error line
306     if (!e) return false;   // thrown null
307     var line = e.lineNumber;
308     if (!line || line == 4294967295) { // lineNumber==maxint in edge cases
309       if (!e.location || !e.location.lineNumber) {
310         GM_logError(e, 0, script._file.uri.spec, 0);
311         return false;
312       }
313       // Sometimes the right one is in "location"
314       line = e.location.lineNumber;
315     }
316     // find problematic file
317     line -= lineRef;
318     var end, start = 1;
319     var uri   = null;
320     for (var i in script.offsets) {
321       end = script.offsets[i];
322       if (line < end) {
323         uri = script.requires[i]._file.uri.spec;
324         break;
325       }
326       start = end;
327     }
328     line -= start;
329     if (!uri)
330       uri = script._file.uri.spec;
331     // log error
332     GM_logError(e, 0, uri, line);
333     return false;
334   }
