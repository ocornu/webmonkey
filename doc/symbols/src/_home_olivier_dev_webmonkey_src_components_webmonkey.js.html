 66   /**
 67    * <code>nsIObserver</code> event handler.
 68    * Is called once on application startup to load subscripts.
 69    */
 70   observe: function(aSubject, aTopic, aData) {
 71     if (aTopic != "app-startup") return;
 72     Components.utils.import("resource://webmonkey/config.js");
 73     var loader = Cc["@mozilla.org/moz/jssubscript-loader;1"]
 74                  .getService(Ci.mozIJSSubScriptLoader);
 75     loader.loadSubScript("chrome://global/content/XPCNativeWrapper.js");
 76     loader.loadSubScript("chrome://webmonkey/content/lib/utils.js");
 77   },

 79   /**
 80    * <code>nsIContentPolicy</code> resource loader hook.
 81    * Allow all resources to be loaded except userscripts, which are redirected
 82    * to the script installer.
 83    * @type  int
 84    */
 85   shouldLoad: function(contentType, contentLocation, requestOrigin, context,
 86                        mimeTypeGuess, extra) {
 87     var ret = Ci.nsIContentPolicy.ACCEPT;
 88 
 89     // block content detection of webmonkey by denying GM
 90     // chrome content, unless loaded from chrome
 91     if (requestOrigin && requestOrigin.scheme != "chrome"
 92         && contentLocation.scheme == "chrome"
 93         && contentLocation.host == "webmonkey")
 94       return Ci.nsIContentPolicy.REJECT_SERVER;
 95 
 96     // don't intercept anything when GM is not enabled
 97     if (!GM_getEnabled()) return ret;
 98 
 99     // don't interrupt the view-source: scheme
100     // (triggered if the link in the error console is clicked)
101     if (contentLocation.scheme == "view-source") return ret;
102 
103     if (contentType == Ci.nsIContentPolicy.TYPE_DOCUMENT
104         && contentLocation.spec.match(/\.user\.js$/)) {
105       var winWat = Cc["@mozilla.org/embedcomp/window-watcher;1"]
106                    .getService(Ci.nsIWindowWatcher);
107       if (winWat.activeWindow && winWat.activeWindow.GM_BrowserUI) {
108         winWat.activeWindow.GM_BrowserUI.downloadScript(contentLocation, true);
109         ret = Ci.nsIContentPolicy.REJECT_REQUEST;
110       }
111     }
112 
113     return ret;
114   },

137   /**
138    * <code>DOMContentLoaded</code> event handler.
139    * Inject relevant scripts once DOM is fully loaded.
140    * @param {Window} unsafeWin
141    *        Target DOM window.
142    * @param {ChromeWindow} chromeWin
143    *        Target chrome window.
144    * @param {GM_BrowserUI} gmBrowser
145    *        <code>GM_BrowserUI</code> responsible for this chrome window.
146    */
147   domContentLoaded: function(unsafeWin, chromeWin, gmBrowser) {
148     var safeWin   = new XPCNativeWrapper(unsafeWin);
149     var href      = safeWin.location.href;
150     var scripts   = this.config.getMatchingScripts(
151       function(script) { return script.meta.enabled && script.isRunnable(href); }
152     );
153     // assert there are scripts to inject
154     if (!scripts.length) return;
155 
156     var firebug = getFirebugConsole(safeWin, unsafeWin, chromeWin);
157     for each (var script in scripts)
158       script.inject(unsafeWin, gmBrowser, firebug);
159   }

173 /* FireBug 1.2+ console support */
174 function getFirebugConsole(safeWin, unsafeWin, chromeWin) {
175   try {
176     chromeWin = chromeWin.top;
177     // assert FB is installed
178     if (!chromeWin.Firebug)
179       return null;
180     var fbVersion = parseFloat(chromeWin.Firebug.version);
181     var fbConsole = chromeWin.Firebug.Console;
182     var fbContext = chromeWin.TabWatcher &&
183                     chromeWin.TabWatcher.getContextByWindow(unsafeWin);
184     // assert FB is enabled
185     if (!fbConsole.isEnabled(fbContext))
186       return null;
187 
188     if (fbVersion == 1.2) {
189       // search console handler
190       if (fbContext.consoleHandler)
191         for (var i = 0; i < fbContext.consoleHandler.length; i++)
192           if (fbContext.consoleHandler[i].window == safeWin)
193             return fbContext.consoleHandler[i].handler;
194       var dummyElm = safeWin.document.createElement("div");
195       dummyElm.setAttribute("id", "_firebugConsole");
196       safeWin.document.documentElement.appendChild(dummyElm);
197       chromeWin.Firebug.Console.injector.addConsoleListener(fbContext, safeWin);
198       dummyElm.parentNode.removeChild(dummyElm);
199       return fbContext.consoleHandler.pop().handler;
200     }
201 
202     if (fbVersion == 1.3 || fbVersion == 1.4) {
203       fbConsole.injector.attachIfNeeded(fbContext, unsafeWin);
204       // find active context
205       for (var i=0; i<fbContext.activeConsoleHandlers.length; i++)
206         if (fbContext.activeConsoleHandlers[i].window == unsafeWin)
207           return fbContext.activeConsoleHandlers[i];
208       return null;
209     }
210   } catch (e) {
211     dump('Webmonkey getFirebugConsole() error:\n'+uneval(e)+'\n');
212   }
213   return null;
214 }
