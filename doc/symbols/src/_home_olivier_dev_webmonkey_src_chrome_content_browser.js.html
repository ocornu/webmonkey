 10 /**

 14   if (!aIID.equals(Components.interfaces.nsISupports) &&
 15       !aIID.equals(Components.interfaces.nsISupportsWeakReference) &&
 16       !aIID.equals(Components.interfaces.nsIWebProgressListener))
 17     throw Components.results.NS_ERROR_NO_INTERFACE;

 36 /**
 37  * The browser XUL has loaded. Find the elements we need and set up our
 38  * listeners and wrapper objects.
 39  */
 40 GM_BrowserUI.chromeLoad = function(e) {
 41   // get all required DOM elements
 42   this.tabBrowser = document.getElementById("content");
 43   this.appContent = document.getElementById("appcontent");
 44   this.contextMenu = document.getElementById("contentAreaContextMenu");
 45   this.statusImage = document.getElementById("gm-status-image");
 46   this.statusLabel = document.getElementById("gm-status-label");
 47   this.statusPopup = document.getElementById("gm-status-popup");
 48   this.statusEnabledItem = document.getElementById("gm-status-enabled-item");
 49   this.generalMenuEnabledItem = document.getElementById("gm-general-menu-enabled-item");
 50   this.toolsMenu = document.getElementById("menu_ToolsPopup");
 51   this.bundle = document.getElementById("gm-browser-bundle");
 52 
 53   // seamonkey compat
 54   if (!this.toolsMenu) {
 55     this.toolsMenu = document.getElementById("taskPopup");
 56   }
 57 
 58   // songbird compat
 59   if (!this.appContent && this.tabBrowser) {
 60     this.appContent = this.tabBrowser.parentNode;
 61   }
 62 
 63   // update visual status when enabled state changes
 64   this.enabledWatcher = GM_hitch(this, "refreshStatus");
 65   GM_prefRoot.watch("enabled", this.enabledWatcher);
 66 
 67   // hook various events
 68   GM_listen(this.appContent, "DOMContentLoaded", GM_hitch(this, "contentLoad"));
 69   GM_listen(this.contextMenu, "popupshowing", GM_hitch(this, "contextMenuShowing"));
 70   GM_listen(this.toolsMenu, "popupshowing", GM_hitch(this, "toolsMenuShowing"));
 71 
 72   // listen for clicks on the install bar
 73   Components.classes["@mozilla.org/observer-service;1"]
 74             .getService(Components.interfaces.nsIObserverService)
 75             .addObserver(this, "install-userscript", true);
 76 
 77   // we use this to determine if we are the active window sometimes
 78   this.winWat = Components.classes["@mozilla.org/embedcomp/window-watcher;1"]
 79                           .getService(Components.interfaces.nsIWindowWatcher);
 80 
 81   // this gives us onLocationChange
 82   this.tabBrowser.addProgressListener(this,
 83     Components.interfaces.nsIWebProgress.NOTIFY_LOCATION);
 84 
 85   // update enabled icon
 86   this.refreshStatus();
 87 
 88   // register for notifications from greasemonkey-service about ui type things
 89   this.gmSvc = Components.classes["@webmonkey.info/webmonkey-service;1"]
 90                .getService().wrappedJSObject;
 91 };

 93 /**
 94  * Gets called when a DOMContentLoaded event occurs somewhere in the browser.
 95  * If that document is in in the top-level window of the focused tab, find
 96  * it's menu items and activate them.
 97  */
 98 GM_BrowserUI.contentLoad = function(e) {
 99   var safeWin;
100   var unsafeWin;
101   var href;
102   var commander;
103 
104   if (!GM_getEnabled()) {
105     return;
106   }
107 
108   safeWin = e.target.defaultView;
109   unsafeWin = safeWin.wrappedJSObject;
110   href = safeWin.location.href;
111 
112   if (GM_isGreasemonkeyable(href)) {
113     commander = this.getCommander(unsafeWin);
114 
115     // if this content load is in the focused tab, attach the menuCommaander
116     if (unsafeWin == this.tabBrowser.selectedBrowser.contentWindow) {
117       this.currentMenuCommander = commander;
118       this.currentMenuCommander.attach();
119     }
120 
121     this.gmSvc.domContentLoaded(unsafeWin, window, this);
122 
123     GM_listen(unsafeWin, "pagehide", GM_hitch(this, "contentUnload"));
124   }
125 
126   // Show the greasemonkey install banner if we are navigating to a .user.js
127   // file in a top-level tab.
128   if (href.match(/\.user\.js$/) && safeWin == safeWin.top) {
129     var browser = this.tabBrowser.getBrowserForDocument(safeWin.document);
130     this.showInstallBanner(browser);
131   }
132 };

135 /**
136  * Shows the install banner across the top of the tab that is displayed when
137  * a user selects "show script source" in the install dialog.
138  */
139 GM_BrowserUI.showInstallBanner = function(browser) {
140   var greeting = this.bundle.getString("greeting.msg");
141 
142   if (this.tabBrowser.showMessage) {
143     // Firefox 1.5 and lower
144     this.tabBrowser.showMessage(
145       browser,
146       "chrome://webmonkey/content/icons/icon_small.png",
147       greeting,
148       this.bundle.getString("greeting.btn"),
149       null /* default doc shell */,
150       "install-userscript",
151       null /* no popuup */,
152       "top",
153       true /* show close button */,
154       this.bundle.getString("greeting.btnAccess") /* access key */);
155   } else {
156     // Firefox 2.0+
157     var notificationBox = this.tabBrowser.getNotificationBox(browser);
158 
159     // Remove existing notifications. Notifications get removed
160     // automatically onclick and on page navigation, but we need to remove
161     // them ourselves in the case of reload, or they stack up.
162     for (var i = 0, child; child = notificationBox.childNodes[i]; i++) {
163       if (child.getAttribute("value") == "install-userscript") {
164         notificationBox.removeNotification(child);
165       }
166     }
167 
168     var notification = notificationBox.appendNotification(
169       greeting,
170       "install-userscript",
171       "chrome://webmonkey/content/icons/icon_small.png",
172       notificationBox.PRIORITY_WARNING_MEDIUM,
173       [{
174         label: this.bundle.getString("greeting.btn"),
175         accessKey: this.bundle.getString("greeting.btnAccess"),
176         popup: null,
177         callback: GM_hitch(this, "installCurrentScript")
178       }]
179     );
180   }
181 };

187   this.statusImage.src = "chrome://global/skin/throbber/Throbber-small.gif";
188   this.statusImage.style.opacity = "0.5";
189   this.statusImage.tooltipText = this.bundle.getString("tooltip.loading");

193       GM_BrowserUI.refreshStatus();
194       GM_BrowserUI.hideStatusImmediately();
195       if (install)
196         GM_BrowserUI.showInstallDialog(script);
197       else
198         GM_BrowserUI.showScriptView(script);
199     },
200     function(source, status, statusText) {
201       window.alert("Could not download script:\n"
202                    + uri.spec + "\n\n"
203                    + "Error "+ status + " (" + statusText +")\n");
204     },
205     true);
206 };

293 /**
294  * A content document has unloaded. We need to remove it's menuCommander to
295  * avoid leaking it's memory.
296  */
297 GM_BrowserUI.contentUnload = function(e) {
298   if (e.persisted) {
299     return;
300   }
301 
302   var unsafeWin = e.target.defaultView;
303 
304   // remove the commander for this document
305   var commander = null;
306 
307   // looping over commanders rather than using getCommander because we need
308   // the index into commanders.splice.
309   for (var i = 0; item = this.menuCommanders[i]; i++) {
310     if (item.win == unsafeWin) {
311 
312       if (item.commander == this.currentMenuCommander) {
313         this.currentMenuCommander.detach();
314         this.currentMenuCommander = null;
315       }
316 
317       this.menuCommanders.splice(i, 1);
318 
319       break;
320     }
321   }
322 };

340 GM_BrowserUI.contextMenuShowing = function() {
341   var contextItem = window.document.getElementById("view-userscript");

344   var culprit = document.popupNode;
345 
346   while (culprit && culprit.tagName && culprit.tagName.toLowerCase() != "a") {

348   }
349 
350   contextItem.hidden =
351     contextSep.hidden =
352     !this.getUserScriptLinkUnderPointer();

356 GM_BrowserUI.getUserScriptLinkUnderPointer = function() {
357   var culprit = document.popupNode;
358 
359   while (culprit && culprit.tagName && culprit.tagName.toLowerCase() != "a") {
360      culprit = culprit.parentNode;
361   }
362 
363   if (!culprit || !culprit.href ||
364       !culprit.href.match(/\.user\.js(\?|$)/i)) {
365     return null;
366   }
367 
368   var ioSvc = Components.classes["@mozilla.org/network/io-service;1"]
369                         .getService(Components.interfaces.nsIIOService);
370   var uri = ioSvc.newURI(culprit.href, null, null);
371 
372   return uri;
373 };

390 /**
391  * Helper method which gets the menuCommander corresponding to a given
392  * document
393  */
394 GM_BrowserUI.getCommander = function(unsafeWin) {
395   for (var i = 0; i < this.menuCommanders.length; i++) {
396     if (this.menuCommanders[i].win == unsafeWin) {
397       return this.menuCommanders[i].commander;
398     }
399   }
400 
401   // no commander found. create one and add it.
402   var commander = new GM_MenuCommander(document);
403   this.menuCommanders.push({win:unsafeWin, commander:commander});
404 
405   return commander;
406 };

429 function GM_showPopup(aEvent) {
430   function urlsOfAllFrames(contentWindow) {
431     function collect(contentWindow) {
432       urls = urls.concat(urlsOfAllFrames(contentWindow));
433     }
434     var urls = [contentWindow.location.href];
435     Array.prototype.slice.call(contentWindow.frames).forEach(collect);
436     return urls;
437   }
438 
439   function uniq(a) {
440     var seen = {}, list = [], item;
441     for (var i = 0; i < a.length; i++) {
442       item = a[i];
443       if (!seen.hasOwnProperty(item))
444         seen[item] = list.push(item);
445     }
446     return list;
447   }
448 
449   function scriptsMatching(urls) {
450 
451     function testMatchURLs(script) {
452 
453       function testMatchURL(url) {
454         return script.isRunnable(url);
455       }
456 
457       return urls.some(testMatchURL);
458     }
459 
460     return GM_getConfig().getMatchingScripts(testMatchURLs);
461   }
462 
463   function appendScriptToPopup(script) {
464     var mi = document.createElement("menuitem");
465     mi.setAttribute("label", script.meta.name);
466     mi.script = script;
467     mi.setAttribute("type", "checkbox");
468     mi.setAttribute("checked", script.meta.enabled.toString());
469     popup.insertBefore(mi, tail);
470   }
471 
472   var popup = aEvent.target;
473   var tail = document.getElementById("gm-status-no-scripts-sep");
474 
475   // set the enabled/disabled state
476   GM_BrowserUI.statusEnabledItem.setAttribute("checked", GM_getEnabled());
477 
478   // remove all the scripts from the list
479   for (var i = popup.childNodes.length - 1; i >= 0; i--) {
480     var node = popup.childNodes[i];
481     if (node.script || node.getAttribute("value") == "hack") {
482       popup.removeChild(node);
483     }
484   }
485 
486   var urls = uniq( urlsOfAllFrames( getBrowser().contentWindow ));
487   var runsOnTop = scriptsMatching( [urls.shift()] ); // first url = top window
488   var runsFramed = scriptsMatching( urls ); // remainder are all its subframes
489 
490   // drop all runsFramed scripts already present in runsOnTop
491   for (var i = 0; i < runsOnTop.length; i++) {
492     var j = 0, item = runsOnTop[i];
493     while (j < runsFramed.length) {
494       if (item === runsFramed[j]) {
495         runsFramed.splice(j, 1);
496       } else {
497         j++;
498       }
499     }
500   }
501 
502   // build the new list of scripts
503   if (runsFramed.length) {
504     runsFramed.forEach(appendScriptToPopup);
505     if (runsOnTop.length) { // only add the separator if there is stuff below
506       var separator = document.createElement("menuseparator");
507       separator.setAttribute("value", "hack"); // remove it in the loop above
508       popup.insertBefore(separator, tail);
509     }
510   }
511   runsOnTop.forEach(appendScriptToPopup);
512 
513   var foundInjectedScript = !!(runsFramed.length + runsOnTop.length);
514   document.getElementById("gm-status-no-scripts").collapsed = foundInjectedScript;
515 }

517 /**
518  * Handle clicking one of the items in the popup. Left-click toggles the enabled
519  * state, rihgt-click opens in an editor.
520  */
521 function GM_popupClicked(aEvent) {
522   if (aEvent.button == 0 || aEvent.button == 2) {
523     var script = aEvent.target.script;
524     if (!script) return;
525 
526     if (aEvent.button == 0) // left-click: toggle enabled state
527       script.meta.enabled =! script.meta.enabled;
528     else // right-click: open in editor
529       script.edit(window);
530 
531     closeMenus(aEvent.target);
532   }
533 }
